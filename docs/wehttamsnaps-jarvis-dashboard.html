// Phase 1 — Base Arch Install
# Update system
sudo pacman -Syu

# Install AUR helper
git clone https://aur.archlinux.org/yay.git
cd yay && makepkg -si

# Core WM + compositor
yay -S swayfx niri waybar wofi rofi
yay -S ghostty foot swaylock-effects
yay -S noctalia-shell-git

// Phase 2 — Gaming Stack
# Steam + Proton
yay -S steam proton-ge-custom
yay -S lutris gamemode gamescope
yay -S mangohud vkbasalt

# Wine stack for MO2
yay -S wine-staging winetricks
yay -S protontricks

# AMD GPU optimizations (RX 580)
sudo echo "RADV_PERFTEST=gpl" >> /etc/environment
sudo echo "AMD_VULKAN_ICD=RADV" >> /etc/environment

// Phase 3 — Photography Workflow
# Photography suite
yay -S darktable digikam gimp krita
yay -S rawtherapee inkscape

# Photography Workflow Workspaces:
# WS5 = DigiKam (import)
# WS5 = Darktable (RAW edit)
# WS5 = GIMP (retouch)
# WS5 = Krita (paint/composite)
// Phase 4 — Audio Routing (PipeWire)
# PipeWire stack
yay -S pipewire pipewire-pulse wireplumber
yay -S qpwgraph helvum pavucontrol

# Create virtual sinks for streaming
# Game audio → sink "game_audio"
# Browser audio → sink "browser_audio"
# Discord → sink "comms_audio"
# All → OBS Virtual Cable

pactl load-module module-null-sink sink_name=game_audio
pactl load-module module-null-sink sink_name=browser_audio
pactl load-module module-null-sink sink_name=comms_audio

// Phase 5 — J.A.R.V.I.S. Sounds
# Install sound-system
sudo cp sound-system /usr/local/bin/
sudo chmod +x /usr/local/bin/sound-system

# Create sound directories
sudo mkdir -p /usr/share/wehttamsnaps/sounds/jarvis
sudo mkdir -p /usr/share/wehttamsnaps/sounds/idroid
